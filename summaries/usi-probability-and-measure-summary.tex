\subsection{Conditional Probability and Bayes' Theorem}

\subsubsection{Events}

\emph{Conditional probability}
\[
\mathrm{P}(A\mid B) = \frac{\mathrm{P}(A \cap B)}{\mathrm{P}(B)}
\]

\emph{Bayes' theorem}
\[
\mathrm{P}(A\mid B) = \frac{\mathrm{P}(B\mid A)\,\mathrm{P}(A)}{\mathrm{P}(B)}
\]

\emph{Total probability}

Let $(A_i)_i$ be a partition of $\Omega$. Then
\[
\begin{aligned}
\mathrm{P}(B)
&= \sum_i \mathrm{P}(B \cap A_i) \\
&= \sum_i \mathrm{P}(B\mid A_i)\,\mathrm{P}(A_i).
\end{aligned}
\]

\section{Probability Spaces}

\subsection{Measurable Spaces}

Let $\Omega$ be a \hl{non-empty} set.
We call $\Omega$ the sample space, which specifies the possible outcomes of a random trial.

Based on the \emph{elementary outcomes} $\omega \in \Omega$, one can also consider more complex \emph{events} $A \subseteq \Omega$, which are based on the usual set operations.

If $P$ is a property of elements of $\Omega$, then the set
\[
\{\omega \in \Omega : P(\omega)\} \subset \Omega
\]
is the subset of all elements $\omega \in \Omega$ for which $P$ is true.
Furthermore, for $A \subset \Omega$, we write
\[
A^\complement := \{\omega \in \Omega : \omega \notin A\}
\]
for the \emph{complement} of $A$ in $\Omega$.
The complement of $\Omega$ is the empty set $\varnothing$.
Given two sets $A,B\subset \Omega$, we further define their \emph{union} by  
\[
A \cup B := \{\omega \in \Omega : \omega \in A \text{ or } \omega \in B\},
\]
their \emph{intersection} by
\[
A \cap B := \{\omega \in \Omega : \omega \in A \text{ and } \omega \in B\},
\]
and their \emph{difference} by
\[
A \setminus B := A \cap B^\complement.
\]

The collection of all possible subsets of $\Omega$ is called the \emph{power set} of $\Omega$ and is defined as
\[
2^\Omega := \{ A : A \subseteq \Omega \}.
\]

We are often interested in certain subsets of the sample space.
To work with these subsets, we define a collection of events that is closed under the usual set operations.

\begin{definition}[Algebra]\label{def:algebra}
A set $\mathcal{F}\subseteq 2^\Omega$ is called an \emph{algebra} on $\Omega$ if
\begin{enumerate}[label=(\roman*)]
  \item $\Omega \in \mathcal{F}$ \label{def:algebra-contains-samplespace}
  \item $A \in \mathcal{F} \Rightarrow A^\complement \in \mathcal{F}$ \label{def:algebra-closed-under-complement}
  \item $A,B \in \mathcal{F} \Rightarrow A \cup B \in \mathcal{F}$ \label{def:algebra-closed-under-union}
\end{enumerate}
The sets $A \in \mathcal{F}$ are called \emph{events}.
\end{definition}

The smallest algebra is $\{\varnothing,\Omega\}$.
Since for any algebra $\mathcal{F}\subseteq 2^\Omega$, $2^\Omega$ is the largest algebra.
Note that any algebra is also closed under intersection, since
$A \cap B = (A^\complement \cup B^\complement)^\complement$ (De Morgan).

\begin{example}[Different information $\Rightarrow$ different algebras]\label{ex-info-algebras}
Alice generates two random bits. The sample space therefore is
\[
\Omega = \{00,01,10,11\}.
\]

Alice observes the exact outcome, hence her observable events are all subsets of $\Omega$:
\[
\mathcal{F}_A = 2^\Omega.
\]

Bob is only told the \emph{number of ones} in the two bits. This induces the partition
\[
\Pi_B = \bigl\{\{00\},\{01,10\},\{11\}\bigr\}
\]
of the sample space $\Omega$.
Therefore Bob can only observe events that are unions of blocks of $\Pi_B$, i.e., the algebra generated by $\Pi_B$:
\[
\mathcal{F}_B
=
\{
\varnothing,\{00\},\{01,10\},\{11\},\{00,11\},\{00,01,10\},\{01,10,11\},\Omega
\}.
\]

Cindy is only told whether the bits are \emph{equal or different}. This induces the coarser partition
\[
\Pi_C = \bigl\{\{00,11\},\{01,10\}\bigr\},
\]
hence her observable events form the algebra:
\[
\mathcal{F}_C = \{\varnothing,\{00,11\},\{01,10\},\Omega\}.
\]

We have
\[
\mathcal{F}_A \supset \mathcal{F}_B \supset \mathcal{F}_C
\]
expressing that more information corresponds to a finer partition and thus a larger algebra of observable events.
\end{example}

Often, we encounter situations with infinitely many events and we want to assign probabilities to the case that all of them occur simultaneously.
This is possible, as long as the number of events involved is at most countably infinite.

\begin{definition}[$\sigma$-algebra]\label{def:sigma-algebra}
A set $\mathcal{F}\subseteq 2^\Omega$ is called a \emph{$\sigma$-algebra} on $\Omega$ if
\begin{enumerate}[label=(\roman*)]
  \item $\mathcal{F}$ is an algebra on $\Omega$,\label{item:sigma-algebra-is-algebra}
  \item for any $A_1,A_2,\dots\in\mathcal{F}$ we have $\bigcup_{i=1}^{\infty} A_i \in \mathcal{F}$.\label{item:sigma-algebra-closed-under-countable-union}
\end{enumerate}
The sets in $\mathcal{F}$ are called \hl{measurable} and the tuple $(\Omega,\mathcal{F})$ is called a \emph{measurable space}.
\end{definition}

\begin{remark}
Let $\mathcal{I}$ be an index set, which may be uncountably infinite, and let $A_i\subseteq \Omega$ for each $i\in \mathcal{I}$.
We denote the union of $A_i$ over $i\in\mathcal{I}$ by
\[
  \bigcup_{i\in\mathcal{I}} A_i := \{\omega \in \Omega : \omega \in A_i \text{ for some } i \in \mathcal{I}\}
\]
and the intersection of $A_i$ over $i \in \mathcal{I}$ by
\[
  \bigcap_{i\in\mathcal{I}} A_i := \{\omega \in \Omega : \omega \in A_i \text{ for all } i \in \mathcal{I}\}.
\]
If $\mathcal{I}$ is countable, e.g.\ $\mathcal{I} = \mathbb{N}$, we often write $\bigcup_{i=1}^{\infty}A_i$ and $\bigcap_{i=1}^{\infty} A_i$ to represent the union and intersection of $A_i$ over $i\in \mathcal{I}$, respectively.
\end{remark}

\begin{example}[Co-finite algebra]\label{ex:cofinite-algebra}
Let $\Omega = \mathbb{N}$ and denote by $\mathcal{F}$ the collection of subsets of $\mathbb{N}$ that are finite or co-finite, i.e.,
\[
  \mathcal{F} = \{A \subseteq \mathbb{N} : |A| < \infty \ \text{or}\ |A^\complement| < \infty\}.
\]
Then $\mathcal{F}$ is an algebra, but not a $\sigma$-algebra,
e.g.\ the sequence $\{2\},\{4\},\{6\},\dots$ is contained in $\mathcal{F}$, but $\bigcup_{i=1}^{\infty} \{i\} \notin \mathcal{F}$.
\end{example}

Recall that for sequences of real numbers, the limit inferior and limit superior are defined as
\[
  \liminf_{i\to\infty} x_i := \lim_{n\to\infty} \inf_{i\ge n} x_i
\quad\text{and}\quad
  \limsup_{i\to\infty} x_i := \lim_{n\to\infty} \sup_{i\ge n} x_i.
\]

\begin{definition}\label{def:liminf-limsup-sets}
Let $A_1,A_2,\dots$ be a sequence of subsets of $\Omega$.
The \emph{limit inferior} and \emph{limit superior} of this sequence are defined as
\[
  \liminf_{i\to\infty} A_i := \bigcup_{i=1}^{\infty} \bigcap_{k=i}^{\infty} A_k
\quad\text{and}\quad
  \limsup_{i\to\infty} A_i := \bigcap_{i=1}^{\infty} \bigcup_{k=i}^{\infty} A_k,
\]
respectively.
\end{definition}

If $\omega \in \liminf_{i\to\infty} A_i$, we say that ``$\omega$ is in $A_i$ eventually'', i.e., for all but finitely many $i$, whereas $\omega \in \limsup_{i\to\infty} A_i$ means that ``$\omega$ is in $A_i$ infinitely often'', i.e. for infinitely many $i$.
We have
\begin{equation}\label{eq:liminf-is-subset-of-limsup}
  \liminf_{i\to\infty} A_i \subseteq \limsup_{i\to\infty} A_i
\end{equation}

\hl{The power set $2^\Omega$ is always the largest $\sigma$-algebra on $\Omega$.}
In contrast, the smallest $\sigma$-algebra is given by $\mathcal{F} = \{\varnothing,\Omega\}$.

\begin{theorem}[Generated $\sigma$-algebra]\label{thm:generated-sigma-algebra}
Let $\mathcal{A}\subseteq 2^\Omega$.
If we define the set
\[
\mathcal{M} := \{\mathcal{F} \mid \mathcal{A} \subseteq \mathcal{F} \subseteq 2^\Omega\ \text{and}\ \mathcal{F}\ \text{is a }\sigma\text{-algebra}\},
\]
the smallest $\sigma$-algebra such that $\mathcal{A} \subseteq \sigma(\mathcal{A})$ is given by
\[
  \sigma(\mathcal{A}) := \bigcap_{\mathcal{F} \in \mathcal{M}} \mathcal{F},
\]
which is called the $\sigma$-algebra generated by $\mathcal{A}$.
\end{theorem}

\begin{proof}
$\mathcal{M}$ is non-empty since $\mathcal{A}\subset 2^\Omega$.
By \Cref{def:algebra} \Cref{def:algebra-contains-samplespace}, $\Omega \in \mathcal{F}$ for all $\mathcal{F}\in \mathcal{M}$, which means that $\Omega \in \sigma(\mathcal{A})$.

If $A \in \sigma(\mathcal{A})$, then $A \in \mathcal{F}$ for all $\mathcal{F}\in \mathcal{M}$.
By \Cref{def:algebra} \Cref{def:algebra-closed-under-complement}, this implies $A^\complement \in \mathcal{F}$ for all $\mathcal{F}\in \mathcal{M}$ and, hence, $A^\complement \in \sigma(\mathcal{A})$.

Let $A_n \in \sigma(\mathcal{A})$ for all $n \in \mathbb{N}$.
Then $A_n \in \mathcal{F}$ for all $n\in \mathbb{N}$ and each $\mathcal{F}\in \mathcal{M}$.
By \Cref{def:sigma-algebra} \Cref{item:sigma-algebra-closed-under-countable-union}, this implies
\[
  A := \bigcup_{n=1}^{\infty} A_n \in \mathcal{F}
\]
for each such $\mathcal{F}$.
Therefore, $A \in \sigma(\mathcal{A})$.

Thus $\sigma(\mathcal{A})$ is a $\sigma$-algebra.
The minimality is a consequence of its definition.
\end{proof}

\begin{example}
Let $(\Omega_1,\mathcal{F}_1)$ and $(\Omega_2,\mathcal{F}_2)$ denote two measurable spaces (see \Cref{def:sigma-algebra}). The \emph{product $\sigma$-algebra} is defined as
\[
  \mathcal{F}_1 \otimes \mathcal{F}_2 := \sigma\bigl(\{A_1 \times A_2 : A_1 \in \mathcal{F}_1, A_2 \in \mathcal{F}_2\}\bigr).
\]


Note that
\[
\{A_1 \times A_2 : A_1 \in \mathcal{F}_1, A_2 \in \mathcal{F}_2\}
\]
is no $\sigma$-algebra in general.
Intuitively, it corresponds to the fact that while rectangles are closed under intersections the same is not true for unions.
The union of two rectangles in general is not a rectangle.
\[
\begin{tikzpicture}[scale=0.7]
\definecolor{aFill}{HTML}{dbfedd}
\definecolor{aStroke}{HTML}{5bd772}
\definecolor{bFill}{HTML}{f8d7d7}

% Rectangle B ((1,2),(4,4))
\filldraw[bFill, draw=bStroke, opacity=0.5] (1,2) rectangle (4,4);
% Rectangle C ((3,1.5),(6,3.5))
\filldraw[cFill, draw=cStroke, opacity=0.5] (3,1.5) rectangle (6,3.5);
\end{tikzpicture}
\]

Hence the family is not even an algebra, and therefore certainly not a $\sigma$-algebra.
If it were closed under complements, closure under unions would follow by De Morgan's laws, leading to a contradiction.
\end{example}

\subsection{Measures}

Introducing the concept of a \emph{measure}, we can assign volumes to sets.

\begin{definition}
  Let $(\Omega,\mathcal{F})$ be a measurable space (see \Cref{def:sigma-algebra}).
  A set function $\mu : \mathcal{F} \to [0,\infty]$ is called a \emph{measure} if
  \begin{enumerate}[label=(\roman*)]
  \item $\mu(\varnothing) = 0$,\smallskip
  \item for any $A_1,A_2,\dots \in \mathcal{F}$ with $A_i \cap A_j = \varnothing$ for $i \neq j$, there holds the \emph{$\sigma$-additivity}
  \begin{equation}
    \mu\Bigl(\bigcup_{i=1}^{\infty} A_i\Bigr) = \sum_{i=1}^{\infty} \mu(A_i)
    \label{eq:sigma-additivity}
  \end{equation}
  \end{enumerate}
  A measure $\mu$ is called \emph{$\sigma$-finite} if there exists a sequence $A_1,A_2,\dots \in \mathcal{F}$ such that $\Omega = \bigcup_{i=1}^{\infty} A_i$ and $\mu(A_i) < \infty$ for all $i\in \mathbb{N}$.
  It is \emph{finite} if $\mu(\Omega) < \infty$.
  The triple $(\Omega,\mathcal{F},\mu)$ is called a \emph{measure space}.
\end{definition}

\begin{remark}
  If a measure assumes the value $\infty$, we make the convention $\infty + \infty = \infty$ and $\infty + c = \infty$ for any $c \in \mathbb{R}$.
\end{remark}

\begin{example}[Counting measure]
Let $\Omega$ be a countable set and consider the measurable space $(\Omega,2^\Omega)$.
We define the set function $\mu$ which assigns a set $A \in \mathcal{F}$ its cardinality, i.e., $\mu(A) = |A|$.
The set function $\mu$ is a measure called the \emph{counting measure}.
\end{example}

To assign \emph{probabilities} to events, we consider the following specialization of a measure.

\begin{definition}[Probability space]\label{def:probability-space}
  Let $(\Omega,\mathcal{F},\mu)$ be a measure space.
  If $\mu(\Omega) = 1$, we call $(\Omega,\mathcal{F},\mu)$ a \emph{probability space} and $\mu$ a \emph{probability measure}.
  In this case, we usually denote $\mu$ by $\Prob$.
\end{definition}

\begin{example}[Discrete probability space]
Let $\Omega = \mathbb{N}$.
Suppose $p_1,p_2,\dots$ are a sequence of nonnegative real numbers such that
\[
  \sum_{i=1}^{\infty} p_i = 1
\]
For any $A \subset \Omega$, we define
\[
\Prob(A) = \sum_{i \in A} p_i
\]
Then, the set function $\Prob: 2^\Omega \to [0,1]$ is a probability measure and $(\Omega,2^\Omega,\Prob)$ is a probability space.
\end{example}

\begin{theorem}\label{thm:subset-measure}
Let $(\Omega,\mathcal{F},\Prob)$ be a measure space.
Let $A,B \in \mathcal{F}$ with $A \subset B$.
Then,
\[
  \mu(A) \le \mu(B)
  \qedhere
\]
\end{theorem}

\begin{proof}
We have $B = A \cup (B \setminus A)$, whereby the sets $A$ and $B \setminus A$ are disjoint.
We infer
\[
  \mu(B) = \mu(A) + \mu(B \setminus A)
\]
and since $\mu(B \setminus A) \ge 0$ by definition, follows that $\mu(B) \ge \mu(A)$.
\end{proof}







\begin{corollary}
Let $(\Omega,\mathcal{F},\mu)$ be a measure space.
Let $A,B \in \mathcal{F}$ with $A \subset B$ and $\mu(A) < \infty$.
Then,
\begin{equation}
  \mu(B \setminus A) = \mu(B) - \mu(A)
  \label{eq:measure-difference}
\end{equation}
\end{corollary}

\begin{theorem}
Let $(\Omega,\mathcal{F},\mu)$ be a measure space and let $A,B \in \mathcal{F}$.
Then,
\[
  \mu(A \cup B) \le \mu(A) + \mu(B)
\]
which can be interpreted as a \emph{triangle inequality}.
\end{theorem}

\begin{proof}
There holds $A \cup B = A \cup (B \setminus A)$, which yields
\[
\mu(A \cup B)
=
\mu(A \cup (B \setminus A))
=
\mu(A) + \mu\underbrace{(B \setminus A)}_{\subset B}
\le
\mu(A) + \mu(B)
\]
by \Cref{thm:subset-measure}.
\end{proof}

\begin{theorem}\label{thm:continuity-from-below}
Let $(\Omega,\mathcal{F},\mu)$ be a measure space and let
\[
A_1 \subset A_2 \subset \dots \in \mathcal{F}
\]
be an increasing sequence of sets.
Then,
\begin{equation}
\boxed{
\lim_{i \to \infty} \mu(A_i)
=
\mu\Bigl(\bigcup_{i=1}^{\infty} A_i\Bigr)
}
\label{eq:continuity-increasing}
\end{equation}
which can be seen as \emph{continuity from below} for sequences of sets.
\end{theorem}

\begin{proof}
We set $B_1 := A_1$ and $B_i := A_i \setminus A_{i-1}$ for $i > 1$.
Then, the sets $B_i$ are mutually disjoint by construction and
$A_i = \bigcup_{j=1}^{i} B_j$.
By $\sigma$-additivity \eqref{eq:sigma-additivity}, we infer
\[\begin{aligned}
  \lim_{i \to \infty} \mu(A_i)
  &=
  \lim_{i \to \infty} \mu\Bigl(\bigcup_{j=1}^{i} B_j\Bigr)
  =
  \lim_{i \to \infty} \sum_{j=1}^{i} \mu(B_j) \\
  &=
  \sum_{j=1}^{\infty} \mu(B_j)
  \stackrel{\eqref{eq:sigma-additivity}}{=}
\mu\Bigl(\bigcup_{j=1}^{\infty} B_j\Bigr)
=
\mu\Bigl(\bigcup_{i=1}^{\infty} A_i\Bigr)
\end{aligned}\]
where we used the fact that $\bigcup_{i=1}^{\infty} B_i = \bigcup_{i=1}^{\infty} A_i$ in the last step.
\end{proof}

\begin{theorem}\label{thm:continuity-from-above}
Let $(\Omega,\mathcal{F},\mu)$ be a measure space and let
\[
A_1 \supset A_2 \supset \dots \in \mathcal{F}
\]
be a decreasing sequence of sets.
Further assume that $\mu(A_{i_0}) < \infty$ for some $i_0 \in \N$.
Then,
\begin{equation}
\boxed{
  \lim_{i \to \infty} \mu(A_i)
  =
  \mu\Bigl(\bigcap_{i=1}^{\infty} A_i\Bigr)
}
\label{eq:continuity-decreasing}
\end{equation}
which can be seen as \emph{continuity from above} for sequences of sets.
\end{theorem}

\begin{proof}
If $i_0 = 1$,
set $B_i := A_1 \setminus A_i$ for $i = 1,2,\dots$.
The sequence $B_1 \subset B_2 \subset \dots$ is increasing and satisfies
\begin{equation}
\mu(B_i)
=
\mu(A_1) - \mu(A_i)
\label{eq:measure-B_i-eq-A_1-minus-A_i}
\end{equation}
by \eqref{eq:measure-difference} since $A_i \subset A_1$.
In addition, we have
\begin{align}
  A_1 \setminus \Bigl(\bigcup_{i=1}^{\infty} B_i\Bigr)
  &=
  A_1 \cap \Bigl(\bigcup_{i=1}^{\infty} B_i\Bigr)^\complement
  =
  A_1 \cap \Bigl(\bigcap_{i=1}^{\infty} B_i^\complement\Bigr)
  =
  \bigcap_{i=1}^{\infty} (A_1 \cap B_i^\complement) \notag\\
  &=
  \bigcap_{i=1}^{\infty} \bigl(A_1 \cap (A_1 \cap A_i^\complement)^\complement\bigr)
  =
  \bigcap_{i=1}^{\infty} (A_1 \cap A_i)
  =
  \bigcap_{i=1}^{\infty} A_i
  \label{eq:A_1-without-union-B_i}
\end{align}
which can be visualized as follows:
\[
\begin{tikzpicture}[x=5mm,y=5mm]
\definecolor{limitFill}{HTML}{\eat #c6eff7}
\definecolor{unionFill}{HTML}{\eat #d5b9e7}
\coordinate (Lc) at (4.0,4.0);
\def\R{2.8}
\coordinate (holeoff) at (-1.0,0.8);
\coordinate (holec) at ($(Lc)+(holeoff)$);
\def\holer{1.0}
\coordinate (cTwo) at ($(Lc)+0.35*(holeoff)$);
\coordinate (cThree) at ($(Lc)+0.65*(holeoff)$);
\draw[line width=0.8pt] (Lc) circle (\R);
\draw[line width=0.8pt] (cTwo) circle (2.2);
\draw[line width=0.8pt] (cThree) circle (1.6);
\filldraw[fill=limitFill,draw=black,line width=0.8pt] (holec) circle (\holer);
\node[font=\fontsize{7pt}{7.5pt}\selectfont] at (5.5,2.3) {$A_1$};
\node[font=\fontsize{7pt}{7.5pt}\selectfont] at (4.7,3.0) {$A_2$};
\node[font=\fontsize{7pt}{7.5pt}\selectfont] at (4.1,3.75) {$\ddots$};

\end{tikzpicture}
\]

Thus,
\[
\begin{aligned}
\lim_{i \to \infty} \mu(A_i)
&=
\mu(A_1) - \bigl(\mu(A_1) - \lim_{i \to \infty} \mu(A_i)\bigr)
=
\mu(A_1) - \lim_{i \to \infty}\bigl(\mu(A_1) - \mu(A_i)\bigr) \\
&\stackrel{\eqref{eq:measure-B_i-eq-A_1-minus-A_i}}{=}
\mu(A_1) - \lim_{i \to \infty} \mu(B_i)
\stackrel{\eqref{eq:continuity-increasing}}{=}
\mu(A_1) - \mu\Bigl(\bigcup_{i=1}^{\infty} B_i\Bigr)
\stackrel{\eqref{eq:measure-difference}}{=}
\mu\Bigl(A_1 \setminus \bigl(\bigcup_{i=1}^{\infty} B_i\bigr)\Bigr) \\
&\stackrel{\eqref{eq:A_1-without-union-B_i}}{=}
\mu\Bigl(\bigcap_{i=1}^{\infty} A_i\Bigr)
\end{aligned}
\]


If $i_0>1$, we apply the same argument to the tail sequence starting at $i_0$.
Since removing finitely many sets from a decreasing sequence does not change the intersection, we have
\(
\bigcap_{i=i_0}^{\infty} A_i
=
\bigcap_{i=1}^{\infty} A_i
\).
Hence,
\(
\lim_{i \to \infty} \mu(A_i)
=
\mu(\bigcap_{i=i_0}^{\infty} A_i)
=
\mu(\bigcap_{i=1}^{\infty} A_i)
\).
\end{proof}

The properties in \Cref{thm:continuity-from-below,thm:continuity-from-above} are known as continuity from below and from above.
If we want to indicate that $(A_i)_{i=1}^{\infty}$ is an increasing sequence of sets with union $A$, we write $A_i \nearrow A$.
To indicate that it is a decreasing sequence of sets with intersection $A$, we write $A_i \searrow A$.

We have the following unifying result.
\begin{theorem}
Let $(\Omega,\mathcal{F},\mu)$ be a measure space and let $A_1,A_2,\dots \in \mathcal{F}$ be a sequence of sets.
Suppose that 
\[
\liminf_{i\to\infty} A_i = \limsup_{i\to\infty} A_i
\]
and denote that set by $\smash[b]{\lim\limits_{i\to\infty} A_i}$.
Then, if $\mu(\bigcup_{i = i_0}^{\infty} A_i) < \infty$ for some $i_0 \in \mathbb{N}$, we have
\[
\lim_{i \to \infty} \mu(A_i) = \mu\Big(\lim_{i\to\infty} A_i\Big)
\]
More generally, if $\mu(\bigcup_{i = i_0}^{\infty} A_i) < \infty$ for some $i_0 \in \mathbb{N}$, we have
\[
\begin{verticalhack}
{\color{red}\boxed{\color{black}
\mu\Big(\liminf_{i\to\infty} A_i\Big)
\leq
\liminf_{i\to\infty} \mu(A_i)
\leq
\limsup_{i\to\infty} \mu(A_i)
\leq
\mu\Big(\limsup_{i\to\infty} A_i\Big)
}}
\end{verticalhack}
\qedhere
\] 
\end{theorem}
\begin{proof}
We prove the general statement, which implies the first one in the case that the limit inferior and limit superior coincide.

A general property for any sequence of real numbers is that $\liminf_{i\to\infty} x_i \leq \limsup_{i\to\infty} x_i$, so the central `$\leq$' in the statement is satisfied.

For the remaining inequalities, we introduce
\[
B_i \coloneq \bigcap_{k=i}^{\infty} A_k
\qquad
C_i \coloneq \bigcup_{k=i}^{\infty} A_k
\]
Note that $\{B_i\}_{i=1}^{\infty}$ is an increasing, while $\{C_i\}_{i=1}^{\infty}$ is a decreasing sequence of sets.
By \autoref{def:liminf-limsup-sets}, we have 
\[
\liminf_{i\to\infty} A_i = \bigcup_{i=1}^{\infty} B_i
\qquad
\limsup_{i\to\infty} A_i = \bigcap_{i=1}^{\infty} C_i
\]

Left `$\leq$':
For $k \geq i$, we have $B_i \subseteq A_k$, and hence, by \autoref{thm:subset-measure}, $\mu(B_i) \leq \mu(A_k)$, whenever $k \geq i$.
This yields
\[
\mu(B_i) \leq \inf_{k \geq i} \mu(A_k)
\]
Taking the limit $i \to \infty$ and using \autoref{thm:continuity-from-below}, we obtain
\[
\mu\Bigl(\liminf_{i\to\infty} A_i\Bigr)
\overset{\text{\hyperref[def:liminf-limsup-sets]{def}}}{=}
\mu\Bigl(\bigcup_{i=1}^{\infty} B_i\Bigr)
\overset{\text{\eqref{eq:continuity-increasing}}}{=}
\lim_{i \to \infty} \mu(B_i)
\leq
\lim_{i \to \infty} \inf_{k \geq i} \mu(A_k)
\overset{\text{def}}{=}
\liminf_{i\to\infty} \mu(A_i)
\]


Right `$\leq$':
For $k \geq i$, we have $C_i \supseteq A_k$, and hence, by \autoref{thm:subset-measure}, $\mu(C_i) \geq \mu(A_k)$, whenever $k \geq i$.
This yields
\[
\mu(C_i) \geq \sup_{k \geq i} \mu(A_k)
\]
Taking the limit $i \to \infty$ and using \autoref{thm:continuity-from-above}, we obtain
\[
\mu\Bigl(\limsup_{i\to\infty} A_i\Bigr)
\overset{\text{\hyperref[def:liminf-limsup-sets]{def}}}{=}
\mu\Bigl(\bigcap_{i=1}^{\infty} C_i\Bigr)
\overset{\text{\eqref{eq:continuity-decreasing}}}{=}
\lim_{i \to \infty} \mu(C_i)
\geq
\lim_{i \to \infty} \sup_{k \geq i} \mu(A_k)
\overset{\text{def}}{=}
\limsup_{i\to\infty} \mu(A_i)
\]

This concludes the proof.
\end{proof}




To conclude this section, we remark that is impossible to define a probability measure that models the unifrom distribution for all subsets of $[0,1]$.
A non-measurable subset is, for example, the \emph{Vitali set}.
\begin{fact}
There is no \emph{translation invariant} measure $\mu : 2^{[0,1]} \to [0,1]$, i.e.,
\[
\mu(A + x) = \mu(A)
\]
for all $A \subseteq [0,1]$ with $A + x \in [0,1]$, $x \in \R$ such that
\[
\mu([a,b]) = b - a
\]
for all $a < b$.
\end{fact}

\begin{exercise}[Borel $\sigma$-algebra on $\mathbb{R}$]
  Let $\mathcal{A}_1 = \{(-\infty,a] : a \in \mathbb{R}\}$ and $\mathcal{A}_2 = \{(-\infty,a) : a \in \mathbb{R}\}$.
  Show that $\sigma(\mathcal{A}_1) = \sigma(\mathcal{A}_2)$.
\end{exercise}

\begin{solution}
  We show both directions of inclusion.

  \emph{Step 1: Show $\mathcal{A}_2 \subset \sigma(\mathcal{A}_1)$.}

  Fix $a \in \mathbb{R}$. Note that
  \[
    (-\infty,a) = \bigcup_{n=1}^{\infty} \Bigl(-\infty,a - \tfrac{1}{n}\Bigr].
  \]
  Each $(-\infty,a - \tfrac{1}{n}] \in \mathcal{A}_1 \subset \sigma(\mathcal{A}_1)$, and $\sigma$-algebras are closed under countable unions, so the union is in $\sigma(\mathcal{A}_1)$. Hence $(-\infty,a) \in \sigma(\mathcal{A}_1)$.

  Therefore $\mathcal{A}_2 \subset \sigma(\mathcal{A}_1)$, which implies $\sigma(\mathcal{A}_2) \subset \sigma(\mathcal{A}_1)$ by minimality.

  \emph{Step 2: Show $\mathcal{A}_1 \subset \sigma(\mathcal{A}_2)$.}

  Fix $a \in \mathbb{R}$. Note that
  \[
    (-\infty,a] = \bigcap_{n=1}^{\infty} \Bigl(-\infty,a + \tfrac{1}{n}\Bigr).
  \]
  Each $(-\infty,a + \tfrac{1}{n}) \in \mathcal{A}_2 \subset \sigma(\mathcal{A}_2)$. Since $\sigma$-algebras are closed under countable intersections (via De Morgan: $\bigcap E_n = \bigl(\bigcup E_n^c\bigr)^c$), the intersection is in $\sigma(\mathcal{A}_2)$.

  Therefore $\mathcal{A}_1 \subset \sigma(\mathcal{A}_2)$, which implies $\sigma(\mathcal{A}_1) \subset \sigma(\mathcal{A}_2)$ by minimality.

  \emph{Conclusion:} From both inclusions, $\sigma(\mathcal{A}_1) = \sigma(\mathcal{A}_2)$.
\end{solution}
